from flask import Flask, request, jsonify import os import base64 from datetime import datetime app = Flask(__name__) UPLOAD_FOLDER = "uploads" os.makedirs(UPLOAD_FOLDER, exist_ok=True) @app.route("/upload", methods=["POST"]) def upload(): data = request.get_json() lat = data.get("lat") lon = data.get("lon") image_data = data.get("image") # base64 PNG data URL print(f"GPS souřadnice: lat={lat}, lon={lon}") # Oříznout prefix 'data:image/png;base64,' z base64 stringu header = "data:image/png;base64," if image_data.startswith(header): image_data = image_data[len(header):] try: img_bytes = base64.b64decode(image_data) except Exception as e: print("Chyba dekódování obrázku:", e) return jsonify({"status": "error", "message": "invalid image data"}), 400 filename = datetime.now().strftime("photo_%Y%m%d_%H%M%S_%f.png") filepath = os.path.join(UPLOAD_FOLDER, filename) with open(filepath, "wb") as f: f.write(img_bytes) print(f"Uloženo {filename}") return jsonify({"status": "ok"}) if __name__ == "__main__": app.run(debug=True)
